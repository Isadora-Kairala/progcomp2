function exe2() {
    let preco = 5.00;
    let reducao = 0;
    let maxLucro = -Infinity;
    let melhorPreco = 0;
    let melhorQtd = 0;

    const tabela = document.getElementById("tabelaLucros").getElementsByTagName("tbody")[0];
    tabela.innerHTML = ""; // limpa a tabela antes de preencher

    while (preco >= 1.00) {
      let ingressos = 120 + (26 * reducao);
      let receita = preco * ingressos;
      let lucro = receita - 200;

      // Criar nova linha na tabela
      let linha = tabela.insertRow();
      linha.insertCell().textContent = preco.toFixed(2);
      linha.insertCell().textContent = ingressos;
      linha.insertCell().textContent = lucro.toFixed(2);

      // Verifica se esse é o maior lucro
      if (lucro > maxLucro) {
        maxLucro = lucro;
        melhorPreco = preco;
        melhorQtd = ingressos;
      }

      preco -= 0.50;
      reducao++;
    }

    // Mostrar lucro máximo
    document.getElementById("resultadoMaximo").textContent =
      `Lucro máximo: R$ ${maxLucro.toFixed(2)} | Preço ideal: R$ ${melhorPreco.toFixed(2)} | Ingressos vendidos: ${melhorQtd}`;
  }
